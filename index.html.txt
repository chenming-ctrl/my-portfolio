<script>
    const portfolio = {
        'Architecture': { kw: 'building', projects: Array.from({length: 5}, (_, i) => ({ id: `a${i}`, title: `Arch Case 0${i+1}` })) },
        'Research': { kw: 'data', projects: Array.from({length: 5}, (_, i) => ({ id: `r${i}`, title: `Res Project 0${i+1}` })) },
        'Photography': { kw: 'film', projects: Array.from({length: 5}, (_, i) => ({ id: `p${i}`, title: `Series 0${i+1}` })) },
        'Interests': { kw: 'art', projects: Array.from({length: 5}, (_, i) => ({ id: `i${i}`, title: `Interests 0${i+1}` })) },
        'Ongoing': { kw: 'draft', projects: Array.from({length: 5}, (_, i) => ({ id: `o${i}`, title: `WIP 0${i+1}` })) }
    };

    let currentLvl = 1, currentCat = '';

    function showLevel(lvl, cat = '') {
        currentLvl = lvl; if(cat) currentCat = cat;
        document.querySelectorAll('.view-layer').forEach(el => el.classList.remove('active'));
        document.getElementById('lvl4').style.display = 'none';
        
        if(lvl === 1) {
            document.getElementById('lvl1').classList.add('active');
            document.getElementById('mainNav').classList.remove('visible');
            document.documentElement.style.setProperty('--bg-blur', '0px');
            isBlurred = false;
        } else if(lvl === 2) {
            renderNeuralGrid('catGrid', Object.keys(portfolio), (c) => showLevel(3, c), true);
            document.getElementById('lvl2').classList.add('active');
            document.getElementById('mainNav').classList.add('visible');
            document.documentElement.style.setProperty('--bg-blur', '15px');
            isBlurred = true;
        } else if(lvl === 3) {
            renderNeuralGrid('projGrid', portfolio[currentCat].projects, (p) => openDetail(currentCat, p.id), false);
            document.getElementById('lvl3').classList.add('active');
            document.documentElement.style.setProperty('--bg-blur', '30px');
        } else if(lvl === 4) {
            document.getElementById('lvl4').style.display = 'block';
            document.getElementById('lvl4').classList.add('active');
            document.body.style.overflowY = 'auto';
        }
        document.getElementById('escHint').style.opacity = (lvl === 1) ? '0' : '1';
    }

    // 核心渲染函数：支持二级和三级的神经网络化
    function renderNeuralGrid(targetId, data, clickFn, isCat) {
        const grid = document.getElementById(targetId);
        grid.innerHTML = data.map((item, i) => {
            const title = isCat ? item : item.title;
            const seed = isCat ? item : item.id;
            return `
                <div class="stack-container" id="sc-${seed}">
                    <div class="neural-stack">
                        <div class="neural-layer layer-3"><img src="https://picsum.photos/seed/${seed}c/300"></div>
                        <div class="neural-layer layer-2"><img src="https://picsum.photos/seed/${seed}b/300"></div>
                        <div class="neural-layer layer-1"><img src="https://picsum.photos/seed/${seed}a/300"></div>
                    </div>
                    <div class="stack-label">${title}</div>
                </div>
            `;
        }).join('');
        
        // 绑定点击事件
        data.forEach(item => {
            const seed = isCat ? item : item.id;
            document.getElementById(`sc-${seed}`).onclick = () => clickFn(item);
        });
    }

    function openDetail(cat, id) {
        const p = portfolio[cat].projects.find(x => x.id === id);
        document.getElementById('detTitle').innerText = p.title;
        document.getElementById('detImages').innerHTML = `<img src="https://picsum.photos/seed/${id}X/1200/800"><img src="https://picsum.photos/seed/${id}Y/1200/800">`;
        showLevel(4);
    }

    window.addEventListener('keydown', e => {
        if (e.key === 'Escape') {
            if (currentLvl === 4) showLevel(3, currentCat);
            else if (currentLvl === 3) showLevel(2);
            else if (currentLvl === 2) showLevel(1);
        }
    });

    const navLinks = document.getElementById('dynamicNav');
    navLinks.innerHTML = Object.keys(portfolio).map(cat => `<a onclick="showLevel(3, '${cat}')">${cat}</a>`).join('');

    // --- 粒子系统：胶囊形 & 鼠标平滑滤波 ---
    const canvas = document.getElementById('particleCanvas');
    const ctx = canvas.getContext('2d');
    let particles = [], isBlurred = false;

    // 滤波后的鼠标坐标
    let mouse = { x: -1000, y: -1000, targetX: -1000, targetY: -1000, vx: 0, vy: 0 };

    window.addEventListener('mousemove', e => {
        mouse.targetX = e.clientX; mouse.targetY = e.clientY;
    });

    class Particle {
        constructor() {
            this.x = Math.random()*canvas.width; this.y = Math.random()*canvas.height;
            this.baseSize = Math.random()*1.5 + 1; this.size = this.baseSize;
            this.color = ['#4285F4', '#34A853', '#FBBC05', '#EA4335', '#9b59b6'][Math.floor(Math.random()*5)];
            this.v = {x:(Math.random()-0.5)*0.5, y:(Math.random()-0.5)*0.5};
            this.angle = 0; this.stretch = 1;
        }
        update() {
            this.x += this.v.x; this.y += this.v.y;
            if(this.x<0||this.x>canvas.width) this.v.x*=-1;
            if(this.y<0||this.y>canvas.height) this.v.y*=-1;
            
            let dx = mouse.x - this.x; let dy = mouse.y - this.y;
            let d = Math.hypot(dx, dy);
            
            if(d < 160) {
                let f = (160-d)/160;
                this.size = this.baseSize + f * 25;
                // 胶囊拉伸感：速度越快拉得越长
                let speed = Math.hypot(mouse.vx, mouse.vy);
                this.stretch = 1 + (speed * 0.25 * f); 
                this.angle = Math.atan2(mouse.vy, mouse.vx); 
            } else {
                this.size += (this.baseSize - this.size) * 0.1;
                this.stretch += (1 - this.stretch) * 0.1;
            }
            this.draw();
        }
        draw() {
            ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.angle);
            ctx.beginPath();
            // 关键：绘制椭圆胶囊
            ctx.ellipse(0, 0, this.size * this.stretch, this.size, 0, 0, Math.PI*2);
            ctx.fillStyle = this.color; ctx.globalAlpha = isBlurred?0.5:0.8;
            ctx.fill(); ctx.restore();
        }
    }

    function init() {
        canvas.width = window.innerWidth; canvas.height = window.innerHeight;
        particles = Array.from({length: 120}, () => new Particle());
    }

    function animate() {
        ctx.clearRect(0,0,canvas.width, canvas.height);

        // 鼠标位置线性插值 (Lerp) - 过滤微小抖动
        let oldX = mouse.x; let oldY = mouse.y;
        mouse.x += (mouse.targetX - mouse.x) * 0.15; 
        mouse.y += (mouse.targetY - mouse.y) * 0.15;
        mouse.vx = mouse.x - oldX; mouse.vy = mouse.y - oldY;

        particles.forEach(p => p.update());
        requestAnimationFrame(animate);
    }
    init(); animate();
    window.addEventListener('resize', init);
</script>
